Strings and vectors

string_vec = c("my", "name", "is", "jeff")

# Returns TRUE for last one because case sensitive
str_detect(string_vec, "jeff")
## [1] FALSE FALSE FALSE  TRUE
# Returns FALSE for last one because case sensitive
str_detect(string_vec, "Jeff")
## [1] FALSE FALSE FALSE FALSE
str_replace(string_vec, "jeff", "zak")
## [1] "my"   "name" "is"   "zak"
# Replace with empty space
str_replace(string_vec, "e", "")
## [1] "my"  "nam" "is"  "jff"

Note that strings are case-sensitive!

string_vec = c(

Why factors are weird

factor_vec = factor(c("male", "male", "female", "female"))
factor_vec
## [1] male   male   female female
## Levels: female male
# Converts females to 1, males to 2
as.numeric(factor_vec)
## [1] 2 2 1 1
factor_vec = fct_relevel(factor_vec, "male")
factor_vec
## [1] male   male   female female
## Levels: male female
# Converts males to 1, females to 2
as.numeric(factor_vec)
## [1] 1 1 2 2

See? Some weird stuff happens in background if you don’t keep track of what’s going on with your labels and factor level orderings.

marj_df = table_marj %>% 
  select(-contains("P Value")) %>% 
  pivot_longer(
    -State,
    names_to = "age_year",
    values_to = "percent"
  ) %>% 
  # In this case, separate at open parenthesis
  separate(age_year, into = c("age", "year"), "\\(") %>% 
  # Get rid of some extraneous characters and make sure percent is numeric
  mutate(
    year = str_replace(year, "\\)", ""),
    percent = str_replace(percent, "[a-c]$", ""),
    percent = as.numeric(percent)
  ) %>% 
  # Do NOT keep anything in state var that matches these conditions (i.e. regional and national totals)
  filter(!State %in% c("Total U.S.", "Northeast", "Midwest", "South", "West"))
